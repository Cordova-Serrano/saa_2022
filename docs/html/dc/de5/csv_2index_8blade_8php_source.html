236 function getDateFormatted(e) {
237 let date = new Date(e)
238 let year = date.getFullYear()
239 let month = ('0' + (date.getMonth() + 1)).slice(-2)
240 let day = date.getDate()
241 let hour = date.getHours()
242 let min = String(date.getMinutes()).padStart(2, '0')
243
244 //Date Formatted
245 date = day + '/' + month + '/' + year + " " + hour + ":" + min
246
247 return date
248 }
